# 读书网站 - 爬虫后台管理界面设计

## 📌 设计目标

为爬虫功能提供可视化后台管理界面，方便：
- 查看爬虫任务执行情况
- 配置爬虫参数
- 监控抓取数据
- 手动触发爬虫任务

## 🎯 技术实现

### Vue 3 + Element Plus 示例

```vue
<template>
  <div class="crawler-admin">
    <!-- 顶部导航栏 -->
    <el-menu
      :default-active="activeMenu"
      mode="horizontal"
      @select="handleMenuSelect"
    >
      <el-menu-item index="dashboard">首页</el-menu-item>
      <el-menu-item index="tasks">爬虫任务</el-menu-item>
      <el-menu-item index="config">平台配置</el-menu-item>
      <el-menu-item index="data">数据浏览</el-menu-item>
      <el-menu-item index="settings">设置</el-menu-item>
      <div class="user-actions">
        <el-button @click="handleLogout">退出登录</el-button>
      </div>
    </el-menu>

    <!-- 首页 -->
    <div v-if="activeMenu === 'dashboard'" class="dashboard">
      <!-- 统计卡片 -->
      <el-row :gutter="20">
        <el-col :span="8">
          <el-card>
            <template #header>
              <div class="card-header">
                <span>总小说数</span>
              </div>
            </template>
            <div class="stat-value">{{ stats.totalNovels }}</div>
          </el-card>
        </el-col>
        <el-col :span="8">
          <el-card>
            <template #header>
              <div class="card-header">
                <span>今日抓取</span>
              </div>
            </template>
            <div class="stat-value">{{ stats.todayCrawled }}</div>
          </el-card>
        </el-col>
        <el-col :span="8">
          <el-card>
            <template #header>
              <div class="card-header">
                <span>任务成功率</span>
              </div>
            </template>
            <div class="stat-value">{{ stats.successRate }}%</div>
          </el-card>
        </el-col>
      </el-row>

      <!-- 平台统计图表 -->
      <el-card class="chart-card">
        <template #header>
          <div class="card-header">
            <span>各平台小说数量</span>
          </div>
        </template>
        <div class="chart-container">
          <div v-for="platform in platformStats" :key="platform.name">
            <div class="platform-bar">
              <div
                class="bar"
                :style="{ width: (platform.count / maxCount) * 100 + '%' }"
              ></div>
              <span class="platform-name">{{ platform.name }}</span>
              <span class="platform-count">{{ platform.count }}</span>
            </div>
          </div>
        </div>
      </el-card>
    </div>

    <!-- 爬虫任务列表 -->
    <div v-if="activeMenu === 'tasks'" class="tasks">
      <div class="toolbar">
        <el-radio-group v-model="taskStatus" @change="fetchTasks">
          <el-radio-button label="">全部</el-radio-button>
          <el-radio-button label="PENDING">PENDING</el-radio-button>
          <el-radio-button label="RUNNING">RUNNING</el-radio-button>
          <el-radio-button label="SUCCESS">SUCCESS</el-radio-button>
          <el-radio-button label="FAILED">FAILED</el-radio-button>
        </el-radio-group>
        <el-button type="primary" @click="fetchTasks">刷新</el-button>
      </div>

      <el-card
        v-for="task in taskList"
        :key="task.id"
        class="task-card"
      >
        <div class="task-header">
          <span class="platform-name">{{ task.platform }}</span>
          <el-tag :type="getStatusType(task.status)">
            {{ task.status }}
          </el-tag>
        </div>
        <div class="task-info">
          <p>任务名：{{ task.taskName }}</p>
          <p>开始时间：{{ formatDate(task.startTime) }}</p>
          <p v-if="task.endTime">结束时间：{{ formatDate(task.endTime) }}</p>
          <p>
            总数：{{ task.totalCount }}
            成功：{{ task.successCount }}
            失败：{{ task.failedCount }}
          </p>
        </div>
        <div class="task-actions">
          <el-button size="small" @click="viewTaskDetail(task)">
            查看详情
          </el-button>
          <el-button
            v-if="task.status === 'FAILED'"
            size="small"
            type="primary"
            @click="retryTask(task)"
          >
            重新执行
          </el-button>
        </div>
      </el-card>

      <!-- 分页器 -->
      <el-pagination
        v-model:current-page="pagination.page"
        v-model:page-size="pagination.size"
        :total="pagination.total"
        layout="total, sizes, prev, pager, next, jumper"
        @current-change="handlePageChange"
        @size-change="handleSizeChange"
      />
    </div>

    <!-- 平台配置 -->
    <div v-if="activeMenu === 'config'" class="config">
      <el-card
        v-for="config in platformConfigs"
        :key="config.platform"
        class="config-card"
      >
        <template #header>
          <div class="config-header">
            <span>{{ getPlatformName(config.platform) }}</span>
            <el-button type="primary" @click="saveConfig(config)">
              保存
            </el-button>
          </div>
        </template>
        <el-form label-width="100px">
```

```http
GET /api/v1/admin/crawler/stats
```

```http
GET /api/v1/admin/crawler/tasks?page=1&size=20&status=SUCCESS
```

```http
GET /api/v1/admin/crawler/configs
```

```http
PUT /api/v1/admin/crawler/config
Content-Type: application/json

{
  "platform": "ciweimao",

  "enabled": true,

  "maxRetry": 3,
```

```http
Content-Type: application/json

{
```
